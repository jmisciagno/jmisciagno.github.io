Definition

An object is a bundle of properties. More formally, it is a dependent function from properties to a value of some type that is associated with that property.
A statement is a conjunction of statements about an object.
Objects can not be contradictory.
Statements can. They can contain negation.

Code

Definition Object (A : Type) (f : A -> Type) : Type :=
  forall (a : A), f a.

Inductive fact (A : Type) (f : A -> Type) : Type :=
| mk_fact' : forall (a : A), f a -> fact A f.

Notation mk_fact := (mk_fact' _ _).

Inductive Expr (A : Type) (f : A -> Type) : Type :=
| unit' : fact A f -> Expr A f
| not' : Expr A f -> Expr A f                           
| disj' : Expr A f -> Expr A f -> Expr A f
| conj' : Expr A f -> Expr A f -> Expr A f.

Notation unit := (unit' _ _).
Notation not := (not' _ _).
Notation disj := (disj' _ _).
Notation conj := (conj' _ _).

Fixpoint describes {A : Type} {f : A -> Type} (x : Expr A f) (o : Object A f) : Prop :=
  match x with
  | unit a =>
    match a with
    | mk_fact p v => o p = v
    end
  | not a => ~ describes a o
  | disj a b => describes a o \/ describes b o
  | conj a b => describes a o /\ describes b o
  end.
